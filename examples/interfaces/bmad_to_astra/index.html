    Bmad to Astra interface

    PyTao, based on Bmad, can be used to create an Astra lattice. This is a simple example of a FODO lattice in Bmad, converted to Astra

    In[1]:
    Copied! 
    from astra import Astra
from pytao import Tao
import matplotlib.pyplot as plt
%config InlineBackend.figure_format = 'retina'
    from astra import Astra from pytao import Tao import matplotlib.pyplot as plt %config InlineBackend.figure_format = 'retina'

    Create Bmad lattice

    In[2]:
    Copied! 
    LAT = """
no_digested
beginning[beta_a] = 10
beginning[beta_b] = 10
beginning[e_tot] = 10e6
parameter[particle] = electron
parameter[geometry] = open

d1: drift, L = 1
d2: d1
qf: quad, L = 0.1, k1 = 1
qd: qf, k1 = -1

fodo: line = (d1, qf, d2, qd)

lat: line = (2*fodo)

use, lat

"""
LATFILE = 'lat.bmad'
with open(LATFILE, 'w') as f:
    f.write(LAT)
    LAT = """ no_digested beginning[beta_a] = 10 beginning[beta_b] = 10 beginning[e_tot] = 10e6 parameter[particle] = electron parameter[geometry] = open d1: drift, L = 1 d2: d1 qf: quad, L = 0.1, k1 = 1 qd: qf, k1 = -1 fodo: line = (d1, qf, d2, qd) lat: line = (2*fodo) use, lat """ LATFILE = 'lat.bmad' with open(LATFILE, 'w') as f: f.write(LAT)
